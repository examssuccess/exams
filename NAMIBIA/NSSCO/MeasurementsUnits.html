(1.) Complete the following.
(a.) 2 km = ....... m
(b.) ....... mg = 12 g
(c.) 25 cm³ = ....... mL
(d.) 0.25 ton = ....... kg


$ \underline{\text{Unity Fraction Method}} \\[3ex] kilo = 10^3 \\[3ex] (a.) \\[3ex] 2\;km * \dfrac{.......m}{.......km} \\[5ex] = 2\;km * \dfrac{10^3\;m}{1\;km} \\[5ex] = 2 * 1000\;m \\[3ex] = 2000\;m \\[5ex] (b.) \\[3ex] milli = 10^{-3} \\[3ex] 12\;g = ?\;mg \\[3ex] 12\;g * \dfrac{.......mg}{.......g} \\[5ex] = 12\;g * \dfrac{1\;mg}{10^{-3}\;g} \\[5ex] = \dfrac{12}{0.001}\;mg \\[5ex] = 12000\;mg \\[5ex] (c.) \\[3ex] L = dm^3 \\[3ex] centi = 10^{-2} \\[3ex] deci = 10^{-1} \\[3ex] \text{First, we shall convert from cm³ to dm³ (L)} \\[3ex] \text{Then, we convert from L to mL} \\[3ex] 25 * cm * cm * cm * \dfrac{...m}{...cm} * \dfrac{...m}{...cm} * \dfrac{...m}{...cm} * \dfrac{...dm}{...m} * \dfrac{...dm}{...m} * \dfrac{...dm}{...m} \\[5ex] = 25 * cm * cm * cm * \dfrac{10^{-2}\;m}{1\;cm} * \dfrac{10^{-2}\;m}{1\;cm} * \dfrac{10^{-2}\;m}{1\;cm} * \dfrac{1\;dm}{10^{-1}\;m} * \dfrac{1\;dm}{10^{-1}\;m} * \dfrac{1\;dm}{10^{-1}\;m} \\[5ex] = \dfrac{25 * 10^{-2 - 2 - 2}}{10^{-1 - 1 - 1}}\;dm^3 ...Law\;1...Exp \\[5ex] =\dfrac{25 * 10^{-6}}{10^{-3}}\;L \\[5ex] = 25 * 10^{-6 - (-3)}\;L ...Law\;2...Exp \\[5ex] = 25 * 10^{-6 + 3}\; L \\[3ex] = 25 * 10^{-3}\;L \\[5ex] \text{Convert from L to mL} \\[3ex] = 25 * 10^{-3}\;L * \dfrac{.......mL}{.......L} \\[5ex] = 25 * 10^{-3}\;L * \dfrac{1\;mL}{10^{-3}\;L} \\[5ex] = 25\;mL \\[5ex] (d.) \\[3ex] 1\;metric\;ton = 1000\;kg \\[3ex] 0.25\;ton * \dfrac{.......kg}{.......ton} \\[5ex] = 0.25\;ton * \dfrac{1000\;kg}{1\;ton} \\[5ex] = 250\;kg $
